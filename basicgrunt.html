grunt-contrib-concat -- Concatenate files.

    1. npm install grunt-contrib-concat --save-dev

    ├── grunt@0.4.5
    └─┬ grunt-contrib-concat@0.5.1
      ├─┬ chalk@0.5.1
      │ ├── ansi-styles@1.1.0
      │ ├── escape-string-regexp@1.0.3
      │ ├─┬ has-ansi@0.1.0
      │ │ └── ansi-regex@0.2.1
      │ ├── strip-ansi@0.3.0
      │ └── supports-color@0.2.0
      └─┬ source-map@0.3.0
        └── amdefine@1.0.0

    2. configure

        // Project configuration.
        grunt.initConfig({
          concat: {
            js: {
              src: ['js/myscriptone.js','js/myscripttwo.js'],
              dest: 'build/js/scripts.js',
            },
            css: {
              src: ['css/reset.css','css/main.css'],
              dest: 'build/css/style.css',
            },
          },
        });

        // grunt contrib concat
        grunt.loadNpmTasks('grunt-contrib-concat');

    3. run "grunt concat" //files (from specific folder/files) should merge in one file (to build/production folder)

******** end

grunt-contrib-watch

    1. npm install grunt-contrib-watch --save-dev

    ├─┬ grunt@0.4.5
    │ ├─┬ findup-sync@0.1.3
    │ │ └── lodash@2.4.2
    │ └─┬ grunt-legacy-log@0.1.2
    │   ├─┬ grunt-legacy-log-utils@0.1.1
    │   │ └── lodash@2.4.2
    │   └── lodash@2.4.2
    └─┬ grunt-contrib-watch@0.6.1
      ├── async@0.2.10
      ├─┬ gaze@0.5.2
      │ └─┬ globule@0.1.0
      │   └── lodash@1.0.2
      ├── lodash@2.4.2
      └─┬ tiny-lr-fork@0.0.5
        ├── debug@0.7.4
        ├── faye-websocket@0.4.4
        ├─┬ noptify@0.0.3
        │ └── nopt@2.0.0
        └── qs@0.5.6

    2. Configure

    module.exports = function(grunt) {

            // watch
            watch: {
                js: {
                    files: ['js/**/*.js'],
                    tasks: ['concat:js'],
                },
                css: {
                    files: ['css/**/*.css'],
                    tasks: ['concat:css'],
                },
            },
        });

        grunt.loadNpmTasks('grunt-contrib-watch');

    };

    3. run "grunt watch"

******** end
